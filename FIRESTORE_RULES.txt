// ========== FIRESTORE SECURITY RULES ==========
// Hãy thêm rules này vào Firestore Console

rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    // Cho phép users đọc/viết data của chính mình
    match /users/{userId} {
      allow read, write: if request.auth.uid == userId;
      
      // Cho phép users tạo, đọc, cập nhật, xóa cards của chính mình
      match /cards/{cardId=**} {
        allow read, create, update, delete: if request.auth.uid == userId;
      }
    }
  }
}
